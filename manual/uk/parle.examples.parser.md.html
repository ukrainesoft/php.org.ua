($l + $r) . " ";                        break;                    case $prod_sub:                        $l = $p->sigil(0);                        $r = $p->sigil(2);                        echo "$l - $r = " . ($l - $r) . " ";                        break;                    case $prod_mul:                        $l = $p->sigil(0);                        $r = $p->sigil(2);                        echo "$l * $r = " . ($l * $r) .
